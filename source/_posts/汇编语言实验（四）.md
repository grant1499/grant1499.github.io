本实验独立完成。

新建 `test.asm` ，

```assembly
assume cs:codesg,ss:stacksg,ds:datasg

stacksg segment
    dw 0,0,0,0,0,0,0,0
stacksg ends

datasg segment
    db '1. display      '  ;将每个单词的前四个字母改成大写
    db '2. brows        '
    db '3. replace      '
    db '4. modify       '
datasg ends

codesg segment
    start:  mov ax,stacksg
            mov ss,ax
            mov sp,16
            mov ax,datasg
            mov ds,ax
            mov si,0
            mov cx,4
            
    s:      push cx
            mov cx,4
            mov bx,0
            
    s0:     mov al,ds:3[bx+si]
            and al,11011111b
            mov ds:3[bx+si],al
            inc bx
            loop s0

            add si,16
            pop cx
            loop s

            mov ax,4c00h
            int 21h
codesg ends

end start
```
